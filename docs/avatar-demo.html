Usage Example

import { generateCharacterAvatar, AvatarOptions } from './characterAvatarGenerator.js';

// Generate from name (deterministic)
const avatar1 = generateCharacterAvatar('John Doe');

// Generate with custom size
const avatar2 = generateCharacterAvatar('Jane Smith', {}, 128);

// Generate with custom options
const avatar3 = generateCharacterAvatar('Alex Johnson', {
    skinTone: AvatarOptions.skinTones[2],
    hairStyle: 'curly',
    hairColor: AvatarOptions.hairColors[4],
    mouthExpression: 'smile',
    accessory: 'glasses'
});

// Insert into DOM
document.getElementById('avatar').innerHTML = avatar1;
